NAME='hgouwdemo'
LOCATION='AustraliaEast'

# Display Azure CLI version
az --version

# Display Azure account locations
az account list-locations

# Login to Azure
az login

# List all your resource groups in a table
az group list --output table

# Create a resource group
az group create --name $NAME --location $LOCATION

# List the details of the resource group
az group list --query "[?name == $NAME]"

# Create an App Service plan
az appservice plan create --name $NAME --resource-group $NAME --location $LOCATION

# List all plans in a table
az appservice plan list --output table

# Create the web app
az webapp create --name $NAME --resource-group $NAME --plan $LOCATION

# List all apps in a table.
az webapp list --output table

# Deploy code from a GitHub repository to the web app
az webapp deployment source config --name hgouwdemo --resource-group hgouwdemo --repo-url "https://github.com/Azure-Samples/php-docs-hello-world" --branch master --manual-integration

# CURL the web app
curl popupwebapp-hgouw.azurewebsites.net

# Create a storage account
az storage account create --name hgouwdemo --resource-group hgouwdemo --location "Australia East" --kind StorageV2  --sku Standard_LRS 

# Display storage account connection string
az storage account show-connection-string --name hgouwdemo --resource-group hgouwdemo

SET AZURE_STORAGE_CONNECTION_STRING="<Azure Storage Settings>"

# Create storage container
az storage container create --name container1 --public-access blob

# Set storage container permission
az storage container set-permission --name container1 --public-access container

# Upload a file to a storage account container
az storage upload --file c:\temp\test.txt --account-name hgouwdemo --container-name container1 --name test.txt -- type block --auth-mode login

# Create function app
az functionapp create --name hgouwdemo --resource-group hgouwdemo --storage-account hgouwdemo --consumption-plan-location  "AustraliaEast"

# Update the FUNCTIONS_EXTENSION_VERSION setting in the function app
az functionapp config appsettings set --name hgouwdemo --resource-group hgouwdemo --settings FUNCTIONS_EXTENSION_VERSION=~2

# Stop the function app
az functionapp stop --name hgouwdemo --resource-group hgouwdemo

# Delete a resource group
az group delete --name hgouwdemo
